import pandas as pd
import numpy as np

# Load dataset
df = pd.read_csv("covid_vaccine_statewise.csv")

# Display dataset
print("First 5 Rows")
print(df.head())

# Dataset Information
print("\nDataset Info")
print(df.info())

# Statistical Summary
print("\nStatistical Summary")
print(df.describe())

# Null Values
print("\nMissing Values")
print(df.isnull().sum())

# State-wise First Dose
print("\nState-wise First Dose Vaccination")

first_dose = df.groupby("State")["First Dose Administered"].max()

print(first_dose)

# State-wise Second Dose
print("\nState-wise Second Dose Vaccination")

second_dose = df.groupby("State")["Second Dose Administered"].max()

print(second_dose)

# Male Vaccinated
male_vaccinated = df["Male(Individuals Vaccinated)"].max()

print("\nTotal Males Vaccinated")
print(male_vaccinated)

# Female Vaccinated
female_vaccinated = df["Female(Individuals Vaccinated)"].max()

print("\nTotal Females Vaccinated")
print(female_vaccinated)
